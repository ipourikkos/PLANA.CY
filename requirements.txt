-- PLANA.CY market observation schema
-- Run in Supabase SQL Editor before market_ingest.py writes observations.

create table if not exists public.market_observations (
    observation_key text primary key,
    source text not null,
    source_id text not null,
    source_class text not null,
    source_listing_id text,
    source_url text,
    transaction_type text,
    property_type text,
    development_status text,
    bedrooms integer,
    bathrooms integer,
    covered_area_m2 numeric,
    plot_area_m2 numeric,
    asking_price_eur numeric,
    asking_rent_monthly_eur numeric,
    price_per_m2_eur numeric,
    rent_per_m2_month_eur numeric,
    latitude double precision,
    longitude double precision,
    district text,
    municipality text,
    locality text,
    planning_zone text,
    title text,
    first_seen_at timestamptz,
    last_seen_at timestamptz,
    price_changed_at timestamptz,
    original_price_eur numeric,
    current_price_eur numeric,
    confidence text,
    source_adapter text,
    source_engine_version text,
    created_at timestamptz not null default now(),
    updated_at timestamptz not null default now()
);

create index if not exists market_observations_location_idx
    on public.market_observations (district, municipality, locality);
create index if not exists market_observations_type_idx
    on public.market_observations (transaction_type, property_type, development_status);
create index if not exists market_observations_geo_idx
    on public.market_observations (latitude, longitude);
create index if not exists market_observations_source_idx
    on public.market_observations (source_id, source_listing_id);

create table if not exists public.market_ingest_runs (
    id bigint generated by default as identity primary key,
    source_id text not null,
    started_at timestamptz not null default now(),
    finished_at timestamptz,
    parsed_count integer not null default 0,
    saved_count integer not null default 0,
    status text not null default 'started',
    error text,
    engine_version text
);

-- Service-role writes are expected. Keep anonymous/public direct writes disabled.
alter table public.market_observations enable row level security;
alter table public.market_ingest_runs enable row level security;
